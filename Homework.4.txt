<!DOCTYPE html>
<html>
<head>
  <title>Canvas Example</title>
  <style>
    #canvas {
      border: 1px solid brown;
    }
  </style>
</head>
<body>
  <canvas id="canvas" width="350" height="250"></canvas>

  <script>
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');

    
    ctx.fillStyle = 'blue';
    ctx.fillRect(50, 50, 100, 75);  // Draw a rectangle

 
    ctx.beginPath();
    ctx.arc(350, 250, 100, 0, 3 * Math.PI);
    ctx.fillStyle = 'red';
    ctx.fill();                                  // Draw a circle

   
    ctx.font = '24px Arial';
    ctx.fillStyle = 'black';
    ctx.fillText('Hello, Canvas!', 250, 360);
                                                        // Draw text
  </script>
</body>
</html>
